Understanding Hash Tables

Collision Resolution Comparison

🔗 Separate Chaining

Each slot contains a linked list. Colliding elements are appended to the list.

  • ✅ Simple to implement
  • ✅ Works well with high load factors
  • ❌ Extra memory for pointers
  • ❌ Cache-unfriendly

📍 Open Addressing

All elements stored in the table. On collision, probe for next empty slot.

  • ✅ Cache-friendly
  • ✅ No extra memory for pointers
  • ❌ Clustering problems
  • ❌ Degrades at high load factors

Probing Methods

Method Probe Sequence Pros Cons
Linear h(k) + i Simple, cache-friendly Primary clustering
Quadratic h(k) + i² Reduces primary clustering Secondary clustering
Double Hashing h₁(k) + i×h₂(k) Best distribution Two hash functions needed

Code Implementation

class HashTableChaining:
    """Hash Table with Separate Chaining"""
    
    def __init__(self, size=11):
        self.size = size
        self.table = [[] for _ in range(size)]
        self.count = 0
    
    def _hash(self, key):
        return sum(ord(c) for c in str(key)) % self.size
    
    def insert(self, key, value):
        index = self._hash(key)
        # Check if key exists, update value
        for i, (k, v) in enumerate(self.table[index]):
            if k == key:
                self.table[index][i] = (key, value)
                return
        # New key, append to chain
        self.table[index].append((key, value))
        self.count += 1
    
    def search(self, key):
        index = self._hash(key)
        for k, v in self.table[index]:
            if k == key:
                return v
        return None
    
    def delete(self, key):
        index = self._hash(key)
        for i, (k, v) in enumerate(self.table[index]):
            if k == key:
                self.table[index].pop(i)
                self.count -= 1
                return True
        return False
    
    @property
    def load_factor(self):
        return self.count / self.size


class HashTableLinearProbing:
    """Hash Table with Linear Probing"""
    
    def __init__(self, size=11):
        self.size = size
        self.keys = [None] * size
        self.values = [None] * size
        self.deleted = [False] * size  # Tombstone markers
        self.count = 0
    
    def _hash(self, key):
        return sum(ord(c) for c in str(key)) % self.size
    
    def insert(self, key, value):
        if self.count >= self.size:
            raise Exception("Table is full")
        
        index = self._hash(key)
        original = index
        
        while self.keys[index] is not None and self.keys[index] != key:
            if self.deleted[index]:  # Can reuse deleted slot
                break
            index = (index + 1) % self.size
            if index == original:
                raise Exception("Table is full")
        
        if self.keys[index] != key:
            self.count += 1
        
        self.keys[index] = key
        self.values[index] = value
        self.deleted[index] = False
    
    def search(self, key):
        index = self._hash(key)
        original = index
        
        while self.keys[index] is not None or self.deleted[index]:
            if self.keys[index] == key and not self.deleted[index]:
                return self.values[index]
            index = (index + 1) % self.size
            if index == original:
                break
        
        return None
    
    def delete(self, key):
        index = self._hash(key)
        original = index
        
        while self.keys[index] is not None or self.deleted[index]:
            if self.keys[index] == key and not self.deleted[index]:
                self.deleted[index] = True  # Mark as tombstone
                self.count -= 1
                return True
            index = (index + 1) % self.size
            if index == original:
                break
        
        return False
