Understanding Stack vs Queue

🥞 Stack (LIFO)

Last In, First Out - Like a stack of plates

  • push(x): Add to top - O(1)
  • pop(): Remove from top - O(1)
  • peek(): View top - O(1)

Use cases: Undo/Redo, Browser history, Function calls

🎫 Queue (FIFO)

First In, First Out - Like a line of people

  • enqueue(x): Add to rear - O(1)
  • dequeue(): Remove from front - O(1)
  • front(): View front - O(1)

Use cases: Task scheduling, BFS, Print queue

class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        self.items.append(item)
    
    def pop(self):
        if self.is_empty():
            raise IndexError("Stack is empty")
        return self.items.pop()
    
    def peek(self):
        if self.is_empty():
            return None
        return self.items[-1]
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)


class Queue:
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        self.items.append(item)
    
    def dequeue(self):
        if self.is_empty():
            raise IndexError("Queue is empty")
        return self.items.pop(0)
    
    def front(self):
        if self.is_empty():
            return None
        return self.items[0]
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)


# Two-Stack Queue Implementation
class QueueWithStacks:
    def __init__(self):
        self.stack_in = []   # For enqueue
        self.stack_out = []  # For dequeue
    
    def enqueue(self, item):
        self.stack_in.append(item)
    
    def dequeue(self):
        if not self.stack_out:
            # Transfer all from stack_in to stack_out
            while self.stack_in:
                self.stack_out.append(self.stack_in.pop())
        if not self.stack_out:
            raise IndexError("Queue is empty")
        return self.stack_out.pop()
