📊 Linear-Time Sorting Algorithms

Beyond Comparison Sorting

Comparison-based sorts (like QuickSort, MergeSort) have a theoretical lower bound of O(n log n). Linear-time sorting algorithms achieve O(n) by not comparing elements directly, but instead using properties of the data (like digit values or ranges).

Algorithm Time Space Stable Best For
Counting Sort O(n + k) O(k) ✅ Yes Small range integers
Radix Sort O(d × n) O(n + k) ✅ Yes Fixed-length integers/strings
Bucket Sort O(n + k) O(n) ✅ Yes Uniformly distributed data

k = range of values, d = number of digits

def counting_sort(arr):
    """Counting Sort - O(n + k) where k is the range"""
    if not arr:
        return arr
    
    # Find range
    min_val, max_val = min(arr), max(arr)
    k = max_val - min_val + 1
    
    # Count occurrences
    count = [0] * k
    for num in arr:
        count[num - min_val] += 1
    
    # Build cumulative count (positions)
    for i in range(1, k):
        count[i] += count[i - 1]
    
    # Build output array (traverse backwards for stability)
    output = [0] * len(arr)
    for num in reversed(arr):
        count[num - min_val] -= 1
        output[count[num - min_val]] = num
    
    return output


def radix_sort(arr):
    """Radix Sort (LSD) - O(d * n) where d is max digits"""
    if not arr:
        return arr
    
    max_val = max(arr)
    exp = 1  # Current digit position (1s, 10s, 100s, ...)
    
    while max_val // exp > 0:
        # Counting sort by current digit
        count = [0] * 10
        output = [0] * len(arr)
        
        for num in arr:
            digit = (num // exp) % 10
            count[digit] += 1
        
        for i in range(1, 10):
            count[i] += count[i - 1]
        
        for num in reversed(arr):
            digit = (num // exp) % 10
            count[digit] -= 1
            output[count[digit]] = num
        
        arr = output
        exp *= 10
    
    return arr


def bucket_sort(arr, num_buckets=5):
    """Bucket Sort - O(n + k) average case"""
    if not arr:
        return arr
    
    min_val, max_val = min(arr), max(arr)
    bucket_range = (max_val - min_val) / num_buckets + 1
    
    # Create buckets
    buckets = [[] for _ in range(num_buckets)]
    
    # Distribute elements
    for num in arr:
        idx = int((num - min_val) / bucket_range)
        idx = min(idx, num_buckets - 1)  # Handle edge case
        buckets[idx].append(num)
    
    # Sort each bucket and concatenate
    result = []
    for bucket in buckets:
        bucket.sort()  # Can use insertion sort for small buckets
        result.extend(bucket)
    
    return result
