def bfs(graph, start):
    visited = set()
    queue = deque([start])
    result = []
    
    while queue:
        node = queue.popleft()  # FIFO - remove from front
        
        if node not in visited:
            visited.add(node)
            result.append(node)
            
            # Add all unvisited neighbors to queue
            for neighbor in graph[node]:
                if neighbor not in visited:
                    queue.append(neighbor)
    
    return result

# Time: O(V + E), Space: O(V)
def dfs_iterative(graph, start):
    visited = set()
    stack = [start]
    result = []
    
    while stack:
        node = stack.pop()  # LIFO - remove from top
        
        if node not in visited:
            visited.add(node)
            result.append(node)
            
            # Add all unvisited neighbors to stack
            # Reverse to maintain left-to-right order
            for neighbor in reversed(graph[node]):
                if neighbor not in visited:
                    stack.append(neighbor)
    
    return result

# Time: O(V + E), Space: O(V)
def dfs_recursive(graph, node, visited=None):
    if visited is None:
        visited = set()
    
    visited.add(node)
    result = [node]
    
    for neighbor in graph[node]:
        if neighbor not in visited:
            result.extend(dfs_recursive(graph, neighbor, visited))
    
    return result

# Time: O(V + E), Space: O(V) - call stack

BFS vs DFS Comparison

Aspect BFS DFS
Data Structure Queue (FIFO) Stack (LIFO)
Exploration Pattern Level by level (breadth) As deep as possible (depth)
Shortest Path ✅ Yes (unweighted graphs) ❌ No
Memory Usage Higher (stores entire level) Lower (stores current path)
Complete ✅ Yes (finds solution if exists) ✅ Yes (finite graphs)
Use Cases Shortest path, level-order traversal Cycle detection, topological sort
