def kruskal(vertices, edges):
    # Sort edges by weight
    edges.sort(key=lambda e: e[2])
    
    # Initialize Union-Find
    parent = {v: v for v in vertices}
    rank = {v: 0 for v in vertices}
    
    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])  # Path compression
        return parent[x]
    
    def union(x, y):
        px, py = find(x), find(y)
        if px == py:
            return False  # Already same component
        if rank[px] < rank[py]:
            px, py = py, px
        parent[py] = px
        if rank[px] == rank[py]:
            rank[px] += 1
        return True
    
    mst = []
    for u, v, weight in edges:
        if union(u, v):  # If edge doesn't create cycle
            mst.append((u, v, weight))
            if len(mst) == len(vertices) - 1:
                break
    
    return mst

# Time: O(E log E), Space: O(V)
import heapq

def prim(graph, start):
    visited = set()
    mst = []
    # Priority queue: (weight, from_vertex, to_vertex)
    pq = [(0, None, start)]
    
    while pq and len(visited) < len(graph):
        weight, from_v, to_v = heapq.heappop(pq)
        
        if to_v in visited:
            continue
            
        visited.add(to_v)
        if from_v is not None:
            mst.append((from_v, to_v, weight))
        
        # Add edges from new vertex to queue
        for neighbor, edge_weight in graph[to_v]:
            if neighbor not in visited:
                heapq.heappush(pq, (edge_weight, to_v, neighbor))
    
    return mst

# Time: O((V + E) log V), Space: O(V + E)

Kruskal vs Prim

Aspect Kruskal's Prim's
Approach Edge-centric (sort all edges) Vertex-centric (grow from start)
Data Structure Union-Find (Disjoint Set) Priority Queue (Min-Heap)
Time Complexity O(E log E) O((V+E) log V)
Better for Sparse graphs (E ≈ V) Dense graphs (E ≈ V²)
Disconnected Graphs Finds forest (MST for each component) Only spans connected component from start
